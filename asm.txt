import void Print(int i)
{
    console.log(i);
}

export void Run()
{
    ; int i

    int.const 0
    local.set i

    loop{
        // if (!(i < 10)) break
        local.get i
        int.const 10
        int.lt_s
        int.eqz
        if{
            break
        }

        // Print(i * 2 + 1)
        local.get i
        int.const 2
        int.mul
        int.const 1
        int.add
        call Print

        // i++
        local.get i
        int.const 1
        int.add
        local.set i
    }
}
